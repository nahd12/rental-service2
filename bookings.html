function loadMyItems() { const container = document.getElementById('myItemsContainer'); if (!container) return; const currentUser = JSON.parse(localStorage.getItem('currentUser')); if (!currentUser) { container.innerHTML = '
Войдите, чтобы просмотреть свои товары
'; return; } const items = JSON.parse(localStorage.getItem('items')) || []; const bookings = JSON.parse(localStorage.getItem('bookings')) || []; const myItems = items.filter(i => i.ownerId === currentUser.id); if (myItems.length === 0) { container.innerHTML = '
У вас пока нет товаров. Создать объявление
'; return; } container.innerHTML = `
${myItems.map(item => { const itemRequests = bookings.filter(b => b.itemId === item.id); return `

${escapeHtml(item.title) || 'Без названия'}

${item.priceDay || 0} ₽/день | Залог: ${item.deposit || 0} ₽

${escapeHtml((item.description || '').substring(0, 80))}...

${itemRequests.length > 0 ? `
Запросы на аренду (${itemRequests.length})
${itemRequests.map(r => { const start = r.startDate ? new Date(r.startDate).toLocaleDateString('ru-RU') : '—'; const end = r.endDate ? new Date(r.endDate).toLocaleDateString('ru-RU') : '—'; const days = r.days || 1; const total = r.totalPrice || (days * (r.deposit || 0)); return `
${escapeHtml(r.renterName) || 'Пользователь'} ${start} — ${end}
Сумма: ${total} ₽ | Залог: ${r.deposit || 0} ₽
${getStatusText(r.status)} ${r.status === 'pending' ? `
` : ''}
`; }).join('')}
` : '
Нет активных запросов
'}
`; }).join('')}
`; } // Добавьте функцию для защиты от XSS function escapeHtml(str) { if (!str) return ''; return str.replace(/[&<>]/g, function(m) { if (m === '&') return '&'; if (m === '<') return '<'; if (m === '>') return '>'; return m; }); }
